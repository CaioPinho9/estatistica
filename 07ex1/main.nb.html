# CARREGAR O PACOTE
library(data.table)
# LEITURA DA BASE
base =
  fread(
    input = paste0("mobile.csv"),
    header = T,
    na.strings = "NA",
    data.table = FALSE,
    dec = "."
  )


write.csv2(base, "base.csv", row.names = FALSE)
# FORMATACAO
library(dplyr)
base$price_range = recode(base$price_range,
                          `0` = "0:baixo",
                          `1` = "1:médio",
                          `2` = "2:caro",
                          `3` = "3:muito caro")

base$four_g = recode(base$four_g,
                          `0` = "0:não tem",
                          `1` = "1:tem")
